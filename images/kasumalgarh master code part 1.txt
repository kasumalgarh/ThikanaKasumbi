    const li = document.createElement('li');

    div.className = 'member-box';
    div.id = `mem-${member.name.replace(/\s+/g, '_')}`;

    if(member.spouse) { div.innerHTML += `<small style="color:gray; font-size:0.8em;">w/o ${member.spouse}</small>`; }

        const ul = document.createElement('ul');
        member.children.forEach(child => {

        });
        li.appendChild(ul);

                pos = getCursorPos(e);
                x = pos.x - (lens.offsetWidth / 2); y = pos.y - (lens.offsetHeight / 2);

                lens.style.left = x + "px"; lens.style.top = y + "px";
                result.style.backgroundPosition = "-" + (x * cx) + "px -" + (y * cy) + "px";
            }
